```powershell
# Fixes vs v1:
# - Added error handling around registry operations
# - Set-ItemProperty wrapped in try/catch

$registryPath = "HKCU:\Software\Microsoft\Windows\CurrentVersion\Internet Settings\ZoneMap\Domains"

# Zone values: 1 = Intranet, 2 = Trusted Sites, 3 = Internet, 4 = Restricted Sites
$sites = @{
    "intranet.net" = 1
}

foreach ($site in $sites.Keys) {
    $zone     = $sites[$site]
    $sitePath = Join-Path $registryPath $site

    try {
        if (-not (Test-Path $sitePath)) {
            New-Item -Path $sitePath -Force | Out-Null
        }
        Set-ItemProperty -Path $sitePath -Name "*" -Value $zone -ErrorAction Stop
        Write-Host "Assigned '$site' to zone $zone." -ForegroundColor Green
    }
    catch {
        Write-Host "Failed to configure zone for '$site': $($_.Exception.Message)" -ForegroundColor Red
    }
}

Write-Host "Site-to-Zone Assignment List has been configured." -ForegroundColor Cyan

```
